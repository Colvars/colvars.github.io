150 int setup(std::vector<int> const &nx_i,
151 T const &t = T(),
152 size_t const &mult_i = 1)
153 {
154 if (cvm::debug()) {
155 cvm::log("Allocating grid: multiplicity = "+cvm::to_str(mult_i)+
156 ", dimensions = "+cvm::to_str(nx_i)+".\n");
157 }
158
159 mult = mult_i;
160
161 data.clear();
162
163 nx = nx_i;
164 nd = nx.size();
165
166 nxc.resize(nd);
167
168 // setup dimensions
169 nt = mult;
170 for (int i = nd-1; i >= 0; i--) {
171 if (nx[i] <= 0) {
172 cvm::error("Error: providing an invalid number of grid points, "+
173 cvm::to_str(nx[i])+".\n", COLVARS_BUG_ERROR);
174 return COLVARS_ERROR;
175 }
176 nxc[i] = nt;
177 nt *= nx[i];
178 }
179
180 if (cvm::debug()) {
181 cvm::log("Total number of grid elements = "+cvm::to_str(nt)+".\n");
182 }
183
184 data.reserve(nt);
185 data.assign(nt, t);
186
187 return COLVARS_OK;
188 }

261 int init_from_colvars(std::vector<colvar *> const &colvars,
262 size_t mult_i = 1,
263 bool add_extra_bin = false)
264 {
265 if (cvm::debug()) {
266 cvm::log("Reading grid configuration from collective variables.\n");
267 }
268
269 cv = colvars;
270 nd = colvars.size();
271 mult = mult_i;
272
273 size_t i;
274
275 if (cvm::debug()) {
276 cvm::log("Allocating a grid for "+cvm::to_str(colvars.size())+
277 " collective variables, multiplicity = "+cvm::to_str(mult_i)+".\n");
278 }
279
280 for (i = 0; i < cv.size(); i++) {
281
282 if (cv[i]->value().type() != colvarvalue::type_scalar) {
283 cvm::error("Colvar grids can only be automatically "
284 "constructed for scalar variables. "
285 "ABF and histogram can not be used; metadynamics "
286 "can be used with useGrids disabled.\n", COLVARS_INPUT_ERROR);
287 return COLVARS_ERROR;
288 }
289
290 if (cv[i]->width <= 0.0) {
291 cvm::error("Tried to initialize a grid on a "
292 "variable with negative or zero width.\n", COLVARS_INPUT_ERROR);
293 return COLVARS_ERROR;
294 }
295
296 widths.push_back(cv[i]->width);
297 hard_lower_boundaries.push_back(cv[i]->is_enabled(colvardeps::f_cv_hard_lower_boundary));
298 hard_upper_boundaries.push_back(cv[i]->is_enabled(colvardeps::f_cv_hard_upper_boundary));
299 periodic.push_back(cv[i]->periodic_boundaries());
300
301 // By default, get reported colvar value (for extended Lagrangian colvars)
302 use_actual_value.push_back(false);
303
304 // except if a colvar is specified twice in a row
305 // then the first instance is the actual value
306 // For histograms of extended-system coordinates
307 if (i > 0 && cv[i-1] == cv[i]) {
308 use_actual_value[i-1] = true;
309 }
310
311 if (add_extra_bin) {
312 if (periodic[i]) {
313 // Shift the grid by half the bin width (values at edges instead of center of bins)
314 lower_boundaries.push_back(cv[i]->lower_boundary.real_value - 0.5 * widths[i]);
315 upper_boundaries.push_back(cv[i]->upper_boundary.real_value - 0.5 * widths[i]);
316 } else {
317 // Make this grid larger by one bin width
318 lower_boundaries.push_back(cv[i]->lower_boundary.real_value - 0.5 * widths[i]);
319 upper_boundaries.push_back(cv[i]->upper_boundary.real_value + 0.5 * widths[i]);
320 }
321 } else {
322 lower_boundaries.push_back(cv[i]->lower_boundary);
323 upper_boundaries.push_back(cv[i]->upper_boundary);
324 }
325 }
326
327 this->init_from_boundaries();
328 return this->setup();
329 }
330
331 int init_from_boundaries()
332 {
333 if (cvm::debug()) {
334 cvm::log("Configuring grid dimensions from colvars boundaries.\n");
335 }
336
337 // these will have to be updated
338 nx.clear();
339 nxc.clear();
340 nt = 0;
341
342 for (size_t i = 0; i < lower_boundaries.size(); i++) {
343 // Re-compute periodicity using current grid boundaries
344 periodic[i] = cv[i]->periodic_boundaries(lower_boundaries[i].real_value,
345 upper_boundaries[i].real_value);
346
347 cvm::real nbins = ( upper_boundaries[i].real_value -
348 lower_boundaries[i].real_value ) / widths[i];
349 int nbins_round = (int)(nbins+0.5);
350
351 if (cvm::fabs(nbins - cvm::real(nbins_round)) > 1.0E-10) {
352 cvm::log("Warning: grid interval("+
353 cvm::to_str(lower_boundaries[i], cvm::cv_width, cvm::cv_prec)+" - "+
354 cvm::to_str(upper_boundaries[i], cvm::cv_width, cvm::cv_prec)+
355 ") is not commensurate to its bin width("+
356 cvm::to_str(widths[i], cvm::cv_width, cvm::cv_prec)+").\n");
357 upper_boundaries[i].real_value = lower_boundaries[i].real_value +
358 (nbins_round * widths[i]);
359 }
360
361 if (cvm::debug())
362 cvm::log("Number of points is "+cvm::to_str((int) nbins_round)+
363 " for the colvar no. "+cvm::to_str(i+1)+".\n");
364
365 nx.push_back(nbins_round);
366 }
367
368 return COLVARS_OK;
369 }

640 inline cvm::real bin_distance_from_boundaries(std::vector<colvarvalue> const &values,
641 bool skip_hard_boundaries = false)
642 {
643 cvm::real minimum = 1.0E+16;
644 for (size_t i = 0; i < nd; i++) {
645
646 if (periodic[i]) continue;
647
648 cvm::real dl = cvm::sqrt(cv[i]->dist2(values[i], lower_boundaries[i])) / widths[i];
649 cvm::real du = cvm::sqrt(cv[i]->dist2(values[i], upper_boundaries[i])) / widths[i];
650
651 if (values[i].real_value < lower_boundaries[i])
652 dl *= -1.0;
653 if (values[i].real_value > upper_boundaries[i])
654 du *= -1.0;
655
656 if ( ((!skip_hard_boundaries) || (!hard_lower_boundaries[i])) && (dl < minimum))
657 minimum = dl;
658 if ( ((!skip_hard_boundaries) || (!hard_upper_boundaries[i])) && (du < minimum))
659 minimum = du;
660 }
661
662 return minimum;
663 }

670 void map_grid(colvar_grid<T> const &other_grid)
671 {
672 if (other_grid.multiplicity() != this->multiplicity()) {
673 cvm::error("Error: trying to merge two grids with values of "
674 "different multiplicity.\n");
675 return;
676 }
677
678 std::vector<colvarvalue> const &gb = this->lower_boundaries;
679 std::vector<cvm::real> const &gw = this->widths;
680 std::vector<colvarvalue> const &ogb = other_grid.lower_boundaries;
681 std::vector<cvm::real> const &ogw = other_grid.widths;
682
683 std::vector<int> ix = this->new_index();
684 std::vector<int> oix = other_grid.new_index();
685
686 if (cvm::debug())
687 cvm::log("Remapping grid...\n");
688 for ( ; this->index_ok(ix); this->incr(ix)) {
689
690 for (size_t i = 0; i < nd; i++) {
691 oix[i] =
692 value_to_bin_scalar(bin_to_value_scalar(ix[i], gb[i], gw[i]),
693 ogb[i],
694 ogw[i]);
695 }
696
697 if (! other_grid.index_ok(oix)) {
698 continue;
699 }
700
701 for (size_t im = 0; im < mult; im++) {
702 this->set_value(ix, other_grid.value(oix, im), im);
703 }
704 }
705
706 has_data = true;
707 if (cvm::debug())
708 cvm::log("Remapping done.\n");
709 }

713 void add_grid(colvar_grid<T> const &other_grid,
714 cvm::real scale_factor = 1.0)
715 {
716 if (other_grid.multiplicity() != this->multiplicity()) {
717 cvm::error("Error: trying to sum togetehr two grids with values of "
718 "different multiplicity.\n");
719 return;
720 }
721 if (scale_factor != 1.0)
722 for (size_t i = 0; i < data.size(); i++) {
723 data[i] += static_cast<T>(scale_factor * other_grid.data[i]);
724 }
725 else
726 // skip multiplication if possible
727 for (size_t i = 0; i < data.size(); i++) {
728 data[i] += other_grid.data[i];
729 }
730 has_data = true;
731 }

782 inline void incr(std::vector<int> &ix) const
783 {
784 for (int i = ix.size()-1; i >= 0; i--) {
785
786 ix[i]++;
787
788 if (ix[i] >= nx[i]) {
789
790 if (i > 0) {
791 ix[i] = 0;
792 continue;
793 } else {
794 // this is the last iteration, a non-valid index is being
795 // set for the outer index, which will be caught by
796 // index_ok()
797 ix[0] = nx[0];
798 return;
799 }
800 } else {
801 return;
802 }
803 }

837 int parse_params(std::string const &conf,
838 colvarparse::Parse_Mode const parse_mode = colvarparse::parse_normal)
839 {
840 if (cvm::debug()) cvm::log("Reading grid configuration from string.\n");
841
842 std::vector<int> old_nx = nx;
843 std::vector<colvarvalue> old_lb = lower_boundaries;
844 std::vector<colvarvalue> old_ub = upper_boundaries;
845 std::vector<cvm::real> old_w = widths;
846
847 {
848 size_t nd_in = 0;
849 // this is only used in state files
850 colvarparse::get_keyval(conf, "n_colvars", nd_in, nd, colvarparse::parse_silent);
851 if (nd_in != nd) {
852 cvm::error("Error: trying to read data for a grid "
853 "that contains a different number of colvars ("+
854 cvm::to_str(nd_in)+") than the grid defined "
855 "in the configuration file("+cvm::to_str(nd)+
856 ").\n");
857 return COLVARS_ERROR;
858 }
859 }
860
861 // underscore keywords are used in state file
862 colvarparse::get_keyval(conf, "lower_boundaries",
