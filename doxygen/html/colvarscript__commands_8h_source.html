15 // The CVSCRIPT macro is used in four distinct contexts:

22 // Each command is created by an instance of the CVSCRIPT macro
23 
24 // The arguments of the CVSCRIPT macro are:
25 
26 // COMM = the id of the command (must be a member of colvarscript::command)
27 
28 // HELP = a one-line description (C string literal) for the command
29 
30 // N_ARGS_MIN = the lowest number of arguments allowed
31 
32 // N_ARGS_MAX = the highest number of arguments allowed
33 
34 // ARGS = multi-line string literal describing each parameter; each line
35 // follows the format "name : type - description"
36 
37 // FN_BODY = the implementation of the function; this should be a thin wrapper
38 // over existing functions; the "script" pointer to the colvarscript
39 // object is already set by the CVSCRIPT_COMM_FN macro; see also the
40 // functions in colvarscript_commands.h.
41

203 CVSCRIPT(cv_list,
204  "Return a list of all variables or biases",
205  0, 1,
206  "param : string - \"colvars\" or \"biases\"; default is \"colvars\"",
207  std::string res;
208  unsigned char *const kwarg = script->get_module_cmd_arg(0, objc, objv);
209  std::string const kwstr = kwarg ? script->obj_to_str(kwarg) :
210  std::string("colvars");
211  if (kwstr == "colvars") {
212  for (std::vector<colvar *>::iterator cvi = script->module()->variables()->begin();
213  cvi != script->module()->variables()->end();
214  ++cvi) {
215  res += (cvi == script->module()->variables()->begin() ? "" : " ") + (*cvi)->name;
216  }
217  script->set_result_str(res);
218  return COLVARS_OK;
219  } else if (kwstr == "biases") {
220  for (std::vector<colvarbias *>::iterator bi = script->module()->biases.begin();
221  bi != script->module()->biases.end();
222  ++bi) {
223  res += (bi == script->module()->biases.begin() ? "" : " ") + (*bi)->name;
224  }
225  script->set_result_str(res);
226  return COLVARS_OK;
227  } else {
228  script->add_error_msg("Wrong arguments to command \"list\"\n");
229  return COLVARSCRIPT_ERROR;
