106 static inline real integer_power(real const &x, int const n)
107 {
108 // Original code: math_special.h in LAMMPS
109 double yy, ww;
110 if (x == 0.0) return 0.0;
111 int nn = (n > 0) ? n : -n;
112 ww = x;
113 for (yy = 1.0; nn != 0; nn >>= 1, ww *=ww) {
114 if (nn & 1) yy *= ww;
115 }
116 return (n > 0) ? yy : 1.0/yy;
117 }
