93 static inline real integer_power(real const &x, int const n)
94 {
95 // Original code: math_special.h in LAMMPS
96 double yy, ww;
97 if (x == 0.0) return 0.0;
98 int nn = (n > 0) ? n : -n;
99 ww = x;
100 for (yy = 1.0; nn != 0; nn >>= 1, ww *=ww) {
101 if (nn & 1) yy *= ww;
102 }
103 return (n > 0) ? yy : 1.0/yy;
104 }
