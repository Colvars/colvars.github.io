79  static inline real integer_power(real const &x, int const n)
80  {
81  // Original code: math_special.h in LAMMPS
82  double yy, ww;
83  if (x == 0.0) return 0.0;
84  int nn = (n > 0) ? n : -n;
85  ww = x;
86  for (yy = 1.0; nn != 0; nn >>= 1, ww *=ww) {
87  if (nn & 1) yy *= ww;
88  }
89  return (n > 0) ? yy : 1.0/yy;
90  }
