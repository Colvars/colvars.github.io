104  static inline real integer_power(real const &x, int const n)
105  {
106  // Original code: math_special.h in LAMMPS
107  double yy, ww;
108  if (x == 0.0) return 0.0;
109  int nn = (n > 0) ? n : -n;
110  ww = x;
111  for (yy = 1.0; nn != 0; nn >>= 1, ww *=ww) {
112  if (nn & 1) yy *= ww;
113  }
114  return (n > 0) ? yy : 1.0/yy;
115  }
