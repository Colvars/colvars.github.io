88  static inline real integer_power(real const &x, int const n)
89  {
90  // Original code: math_special.h in LAMMPS
91  double yy, ww;
92  if (x == 0.0) return 0.0;
93  int nn = (n > 0) ? n : -n;
94  ww = x;
95  for (yy = 1.0; nn != 0; nn >>= 1, ww *=ww) {
96  if (nn & 1) yy *= ww;
97  }
98  return (n > 0) ? yy : 1.0/yy;
99  }
