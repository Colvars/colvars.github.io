32  class n_matrix { // Stores the distribution matrix of n(x,y)
33 
34  public:
35  n_matrix() {}
36  n_matrix(const std::vector<double> & lowerboundary_input, // lowerboundary of x
37  const std::vector<double> & upperboundary_input, // upperboundary of
38  const std::vector<double> & width_input, // width of x
39  const int y_size_input) { // size of y, for example, ysize=7, then when x=1, the distribution of y in [-2,4] is considered
40 
41  int i;
42 
43  this->lowerboundary = lowerboundary_input;
44  this->upperboundary = upperboundary_input;
45  this->width = width_input;
46  this->dimension = lowerboundary_input.size();
47  this->y_size = y_size_input; // keep in mind the internal (spare) matrix is stored in diagonal form
48  this->y_total_size = int(cvm::pow(double(y_size_input), double(dimension)) + EPSILON);
49 
50  // the range of the matrix is [lowerboundary, upperboundary]
51  x_total_size = 1;
52  for (i = 0; i < dimension; i++) {
53  x_size.push_back(int((upperboundary_input[i] - lowerboundary_input[i]) / width_input[i] + EPSILON));
54  x_total_size *= x_size[i];
55  }
56 
57  // initialize the internal matrix
58  matrix.reserve(x_total_size);
59  for (i = 0; i < x_total_size; i++) {
60  matrix.push_back(std::vector<int>(y_total_size, 0));
61  }
62 
63  temp.resize(dimension);
64  }
65 
66  int inline get_value(const std::vector<double> & x, const std::vector<double> & y) {
67  return matrix[convert_x(x)][convert_y(x, y)];
68  }
69 
70  void inline set_value(const std::vector<double> & x, const std::vector<double> & y, const int value) {
71  matrix[convert_x(x)][convert_y(x,y)] = value;
72  }
73 
74  void inline increase_value(const std::vector<double> & x, const std::vector<double> & y, const int value) {
75  matrix[convert_x(x)][convert_y(x,y)] += value;
76  }

78  private:
79  std::vector<double> lowerboundary;
80  std::vector<double> upperboundary;
81  std::vector<double> width;
82  int dimension;
83  std::vector<int> x_size; // the size of x in each dimension
84  int x_total_size; // the size of x of the internal matrix
85  int y_size; // the size of y in each dimension
86  int y_total_size; // the size of y of the internal matrix
87 
88  std::vector<std::vector<int> > matrix; // the internal matrix
89 
90  std::vector<int> temp; // this vector is used in convert_x and convert_y to save computational resource
91 
92  int convert_x(const std::vector<double> & x) { // convert real x value to its interal index
93 
94  int i, j;
95 
96  for (i = 0; i < dimension; i++) {
97  temp[i] = int((x[i] - lowerboundary[i]) / width[i] + EPSILON);
98  }
99 
100  int index = 0;
101  for (i = 0; i < dimension; i++) {
102  if (i + 1 < dimension) {
103  int x_temp = 1;
104  for (j = i + 1; j < dimension; j++)
105  x_temp *= x_size[j];
106  index += temp[i] * x_temp;
107  }
108  else
109  index += temp[i];
110  }
111  return index;
112  }
113 
114  int convert_y(const std::vector<double> & x, const std::vector<double> & y) { // convert real y value to its interal index
115 
116  int i;
117 
118  for (i = 0; i < dimension; i++) {
119  temp[i] = int(round((round(y[i] / width[i] + EPSILON) - round(x[i] / width[i] + EPSILON)) + (y_size - 1) / 2 + EPSILON));
120  }
121 
122  int index = 0;
123  for (i = 0; i < dimension; i++) {
124  if (i + 1 < dimension)
125  index += temp[i] * int(cvm::pow(double(y_size), double(dimension - i - 1)) + EPSILON);
126  else
127  index += temp[i];
128  }
129  return index;
130  }
131 
132  double round(double r) {
133  return (r > 0.0) ? floor(r + 0.5) : ceil(r - 0.5);
134  }
135  };

295  }

303  // called from MD engine every step
304  bool update(cvm::step_number step, std::vector<double> x, std::vector<double> y) {
305 
306  int i;
307 
308  for (i = 0; i < dimension; i++) {
309  // for dihedral RC, it is possible that x = 179 and y = -179, should correct it
310  // may have problem, need to fix
311  if (x[i] > 150 && y[i] < -150) {
312  y[i] += 360;
313  }
314  if (x[i] < -150 && y[i] > 150) {
315  y[i] -= 360;
316  }
317 
318  if (x[i] < lowerboundary[i] - EXTENDED_X_SIZE * width[i] + EPSILON || x[i] > upperboundary[i] + EXTENDED_X_SIZE * width[i] - EPSILON \
319  || y[i] - x[i] < -HALF_Y_SIZE * width[i] + EPSILON || y[i] - x[i] > HALF_Y_SIZE * width[i] - EPSILON \
320  || y[i] - lowerboundary[i] < -HALF_Y_SIZE * width[i] + EPSILON || y[i] - upperboundary[i] > HALF_Y_SIZE * width[i] - EPSILON)
321  return false;
322  }
323 
324  for (i = 0; i < dimension; i++) {
325  sum_x[i].increase_value(y, x[i]);
326  sum_x_square[i].increase_value(y, x[i] * x[i]);
327  }
328  count_y.increase_value(y, 1);
329 
330  for (i = 0; i < dimension; i++) {
331  // adapt colvars precision
332  if (x[i] < lowerboundary[i] + EPSILON || x[i] > upperboundary[i] - EPSILON)
333  return false;
334  }
335  distribution_x_y.increase_value(x, y, 1);
336 
337  return true;
338  }

379  // calculate gradients from the internal variables
380  void calc_pmf() {
381  int norm;
382  int i, j, k;
383 
384  std::vector<double> loop_flag(dimension, 0);
385  for (i = 0; i < dimension; i++) {
386  loop_flag[i] = lowerboundary[i] - HALF_Y_SIZE * width[i];
387  }
388 
389  i = 0;
390  while (i >= 0) {
391  norm = count_y.get_value(loop_flag) > 0 ? count_y.get_value(loop_flag) : 1;
392  for (j = 0; j < dimension; j++) {
393  x_av[j].set_value(loop_flag, sum_x[j].get_value(loop_flag) / norm);
394  sigma_square[j].set_value(loop_flag, sum_x_square[j].get_value(loop_flag) / norm - x_av[j].get_value(loop_flag) * x_av[j].get_value(loop_flag));
395  }
396 
397  // iterate over any dimensions
398  i = dimension - 1;
399  while (i >= 0) {
400  loop_flag[i] += width[i];
401  if (loop_flag[i] > upperboundary[i] + HALF_Y_SIZE * width[i] - width[i] + EPSILON) {
402  loop_flag[i] = lowerboundary[i] - HALF_Y_SIZE * width[i];
403  i--;
404  }
405  else
406  break;
407  }
408  }
409 
410  // double integration
411  std::vector<double> av(dimension, 0);
412  std::vector<double> diff_av(dimension, 0);
413 
414  std::vector<double> loop_flag_x(dimension, 0);
415  std::vector<double> loop_flag_y(dimension, 0);
416  for (i = 0; i < dimension; i++) {
417  loop_flag_x[i] = lowerboundary[i];
418  loop_flag_y[i] = loop_flag_x[i] - HALF_Y_SIZE * width[i];
419  }
420 
421  i = 0;
422  while (i >= 0) {
423  norm = 0;
424  for (k = 0; k < dimension; k++) {
425  av[k] = 0;
426  diff_av[k] = 0;
427  loop_flag_y[k] = loop_flag_x[k] - HALF_Y_SIZE * width[k];
428  }
429 
430  j = 0;
431  while (j >= 0) {
432  norm += distribution_x_y.get_value(loop_flag_x, loop_flag_y);
433  for (k = 0; k < dimension; k++) {
434  if (sigma_square[k].get_value(loop_flag_y) > EPSILON || sigma_square[k].get_value(loop_flag_y) < -EPSILON)
435  av[k] += distribution_x_y.get_value(loop_flag_x, loop_flag_y) * ( (loop_flag_x[k] + 0.5 * width[k]) - x_av[k].get_value(loop_flag_y)) / sigma_square[k].get_value(loop_flag_y);
436 
437  diff_av[k] += distribution_x_y.get_value(loop_flag_x, loop_flag_y) * (loop_flag_x[k] - loop_flag_y[k]);
438  }
439 
440  // iterate over any dimensions
441  j = dimension - 1;
442  while (j >= 0) {
443  loop_flag_y[j] += width[j];
444  if (loop_flag_y[j] > loop_flag_x[j] + HALF_Y_SIZE * width[j] - width[j] + EPSILON) {
445  loop_flag_y[j] = loop_flag_x[j] - HALF_Y_SIZE * width[j];
446  j--;
447  }
448  else
449  break;
450  }
451  }
452 
453  std::vector<double> grad_temp(dimension, 0);
454  for (k = 0; k < dimension; k++) {
455  diff_av[k] /= (norm > 0 ? norm : 1);
456  av[k] = cvm::boltzmann() * temperature * av[k] / (norm > 0 ? norm : 1);
457  grad_temp[k] = av[k] - krestr[k] * diff_av[k];
458  }
459  grad.set_value(loop_flag_x, grad_temp);
460  count.set_value(loop_flag_x, norm);
461 
462  // iterate over any dimensions
463  i = dimension - 1;
464  while (i >= 0) {
465  loop_flag_x[i] += width[i];
466  if (loop_flag_x[i] > upperboundary[i] - width[i] + EPSILON) {
467  loop_flag_x[i] = lowerboundary[i];
468  i--;
469  }
470  else
471  break;
472  }
473  }
474  }

477  // calculate 1D pmf
478  void calc_1D_pmf()
479  {
480  std::vector<double> last_position(1, 0);
481  std::vector<double> position(1, 0);
482 
483  double min = 0;
484  double dG = 0;
485  double i;
486 
487  oneD_pmf.set_value(lowerboundary, 0);
488  last_position = lowerboundary;
489  for (i = lowerboundary[0] + width[0]; i < upperboundary[0] + EPSILON; i += width[0]) {
490  position[0] = i + EPSILON;
491  if (restart == false || input_count.get_value(last_position) == 0) {
492  dG = oneD_pmf.get_value(last_position) + grad.get_value(last_position)[0] * width[0];
493  }
494  else {
495  dG = oneD_pmf.get_value(last_position) + ((grad.get_value(last_position)[0] * count.get_value(last_position) + input_grad.get_value(last_position)[0] * input_count.get_value(last_position)) / (count.get_value(last_position) + input_count.get_value(last_position))) * width[0];
496  }
497  if (dG < min)
498  min = dG;
499  oneD_pmf.set_value(position, dG);
500  last_position[0] = i + EPSILON;
501  }
502 
503  for (i = lowerboundary[0]; i < upperboundary[0] + EPSILON; i += width[0]) {
504  position[0] = i + EPSILON;
505  oneD_pmf.set_value(position, oneD_pmf.get_value(position) - min);
506  }
507  }
