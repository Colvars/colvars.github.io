125 extended:
126 H = H_{0} + \sum_{i} 1/2*\lambda*(S_i(x(t))-s_i(t))^2 \\
127 + \sum_{i} 1/2*m_i*(ds_i(t)/dt)^2 \\
128 + \sum_{t'<t} W * exp(-1/2*\sum_{i} (s_i(t')-s_i(t))^2/(\delta{}s_i)^2) \\
129 + \sum_{w} (\sum_{i}c_{w,i}s_i(t) - D_w)^M/(\Sigma_w)^M
130
131 normal:
132 H = H_{0} + \sum_{t'<t} W * exp(-1/2*\sum_{i} (S_i(x(t'))-S_i(x(t)))^2/(\delta{}S_i)^2) \\
133 + \sum_{w} (\sum_{i}c_{w,i}S_i(t) - D_w)^M/(\Sigma_w)^M
134
135 output:
136 H = H_{0} (only output S(x), no forces)
137
138 Here:
139 S(x(t)) = x
140 s(t) = x_ext
141 DS = Ds = delta
142 */

161 inline colvarvalue fdiff_velocity(colvarvalue const &xold,
162 colvarvalue const &xnew)
163 {
164 // using the gradient of the square distance to calculate the
165 // velocity (non-scalar variables automatically taken into
166 // account)
167 cvm::real dt = cvm::dt();
168 return ( ( (dt > 0.0) ? (1.0/dt) : 1.0 ) *
169 0.5 * dist2_lgrad(xnew, xold) );
170 }
