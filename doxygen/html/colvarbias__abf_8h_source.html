116 inline int update_system_force(size_t i)
117 {
118 if (colvars[i]->is_enabled(f_cv_subtract_applied_force)) {
119 // this colvar is already subtracting the ABF force
120 system_force[i] = colvars[i]->total_force().real_value;
121 } else {
122 system_force[i] = colvars[i]->total_force().real_value
123 - colvar_forces[i].real_value;
124 // If hideJacobian is active then total_force has an extra term of -fj
125 // which is the Jacobian-compensating force at the colvar level
126 }
127 if (cvm::debug())
128 cvm::log("ABF System force calc: cv " + cvm::to_str(i) +
129 " fs " + cvm::to_str(system_force[i]) +
130 " = ft " + cvm::to_str(colvars[i]->total_force().real_value) +
131 " - fa " + cvm::to_str(colvar_forces[i].real_value));
132 return COLVARS_OK;
133 }
