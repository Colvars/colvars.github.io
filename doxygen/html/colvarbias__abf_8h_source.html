113 inline int update_system_force(size_t i)
114 {
115 if (colvars[i]->is_enabled(f_cv_subtract_applied_force)) {
116 // this colvar is already subtracting the ABF force
117 system_force[i] = colvars[i]->total_force().real_value;
118 } else {
119 system_force[i] = colvars[i]->total_force().real_value
120 - colvar_forces[i].real_value;
121 // If hideJacobian is active then total_force has an extra term of -fj
122 // which is the Jacobian-compensating force at the colvar level
123 }
124 if (cvm::debug())
125 cvm::log("ABF System force calc: cv " + cvm::to_str(i) +
126 " fs " + cvm::to_str(system_force[i]) +
127 " = ft " + cvm::to_str(colvars[i]->total_force().real_value) +
128 " - fa " + cvm::to_str(colvar_forces[i].real_value));
129 return COLVARS_OK;
130 }
