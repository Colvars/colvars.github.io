25 std::istream & colvar_grid<T>::read_multicol(std::istream &is, bool add)
26 {
27  // Data in the header: nColvars, then for each
28  // xiMin, dXi, nPoints, periodic flag
29 
30  std::string hash;
31  cvm::real lower, width, x;
32  size_t n, periodic_flag;
33  bool remap;
34  std::vector<T> new_value;
35  std::vector<int> nx_read;
36  std::vector<int> bin;
37 
38  if ( cv.size() > 0 && cv.size() != nd ) {
39  cvm::error("Cannot read grid file: number of variables in file differs from number referenced by grid.\n");
40  return is;
41  }
42 
43  if ( !(is >> hash) || (hash != "#") ) {
44  cvm::error("Error reading grid at position "+
45  cvm::to_str(static_cast<size_t>(is.tellg()))+
46  " in stream(read \"" + hash + "\")\n", COLVARS_INPUT_ERROR);
47  return is;
48  }
49 
50  is >> n;
51  if ( n != nd ) {
52  cvm::error("Error reading grid: wrong number of collective variables.\n");
53  return is;
54  }
55 
56  nx_read.resize(n);
57  bin.resize(n);
58  new_value.resize(mult);
59 
60  if (this->has_parent_data && add) {
61  new_data.resize(data.size());
62  }
63 
64  remap = false;
65  for (size_t i = 0; i < nd; i++ ) {
66  if ( !(is >> hash) || (hash != "#") ) {
67  cvm::error("Error reading grid at position "+
68  cvm::to_str(static_cast<size_t>(is.tellg()))+
69  " in stream(read \"" + hash + "\")\n");
70  return is;
71  }
72 
73  is >> lower >> width >> nx_read[i] >> periodic_flag;
74 
75 
76  if ( (cvm::fabs(lower - lower_boundaries[i].real_value) > 1.0e-10) ||
77  (cvm::fabs(width - widths[i] ) > 1.0e-10) ||
78  (nx_read[i] != nx[i]) ) {
79  cvm::log("Warning: reading from different grid definition (colvar "
80  + cvm::to_str(i+1) + "); remapping data on new grid.\n");
81  remap = true;
82  }
83  }
84 
85  if ( remap ) {
86  // re-grid data
87  while (is.good()) {
88  bool end_of_file = false;
89 
90  for (size_t i = 0; i < nd; i++ ) {
91  if ( !(is >> x) ) end_of_file = true;
92  bin[i] = value_to_bin_scalar(x, i);
93  }
94  if (end_of_file) break;
95 
96  for (size_t imult = 0; imult < mult; imult++) {
97  is >> new_value[imult];
98  }
99 
100  if ( index_ok(bin) ) {
101  for (size_t imult = 0; imult < mult; imult++) {
102  value_input(bin, new_value[imult], imult, add);
103  }
104  }
105  }
106  } else {
107  // do not re-grid the data but assume the same grid is used
108  for (std::vector<int> ix = new_index(); index_ok(ix); incr(ix) ) {
109  for (size_t i = 0; i < nd; i++ ) {
110  is >> x;
111  }
112  for (size_t imult = 0; imult < mult; imult++) {
113  is >> new_value[imult];
114  value_input(ix, new_value[imult], imult, add);
115  }
116  }
117  }
118  has_data = true;
119  return is;
120 }
