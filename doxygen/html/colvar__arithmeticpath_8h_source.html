70template <typename element_type, typename scalar_type, path_sz path_type>
71void ArithmeticPathBase<element_type, scalar_type, path_type>::computeValue() {
72 updateDistanceToReferenceFrames();
73 numerator_s = scalar_type(0);
74 denominator_s = scalar_type(0);
75 for (size_t i_frame = 0; i_frame < frame_element_distances.size(); ++i_frame) {
76 scalar_type exponent_tmp = scalar_type(0);
77 for (size_t j_elem = 0; j_elem < num_elements; ++j_elem) {
78 exponent_tmp += weights[j_elem] * frame_element_distances[i_frame][j_elem] * weights[j_elem] * frame_element_distances[i_frame][j_elem];
79 }
80 exponent_tmp = exponent_tmp * -1.0 * lambda;
81 // prevent underflow if the argument of cvm::exp is less than -708.4
82 if (exponent_tmp > -708.4) {
83 exponent_tmp = cvm::exp(exponent_tmp);
84 } else {
85 exponent_tmp = 0;
86 }
87 numerator_s += static_cast<scalar_type>(i_frame) * exponent_tmp;
88 denominator_s += exponent_tmp;
89 s_numerator_frame[i_frame] = static_cast<scalar_type>(i_frame) * exponent_tmp;
90 s_denominator_frame[i_frame] = exponent_tmp;
91 }
92 s = numerator_s / denominator_s * normalization_factor;
93 z = -1.0 / lambda * cvm::logn(denominator_s);
94}

103void ArithmeticPathBase<element_type, scalar_type, path_type>::computeDerivatives() {
104 for (size_t j_elem = 0; j_elem < num_elements; ++j_elem) {
105 element_type dsdxj_numerator_part1(dsdx[j_elem]);
106 element_type dsdxj_numerator_part2(dsdx[j_elem]);
107 element_type dzdxj_numerator(dsdx[j_elem]);
108 dsdxj_numerator_part1.reset();
109 dsdxj_numerator_part2.reset();
110 dzdxj_numerator.reset();
111 for (size_t i_frame = 0; i_frame < frame_element_distances.size(); ++i_frame) {
112 element_type derivative_tmp = -2.0 * lambda * weights[j_elem] * weights[j_elem] * frame_element_distances[i_frame][j_elem];
113 dsdxj_numerator_part1 += s_numerator_frame[i_frame] * derivative_tmp;
114 dsdxj_numerator_part2 += s_denominator_frame[i_frame] * derivative_tmp;
115 dzdxj_numerator += s_denominator_frame[i_frame] * derivative_tmp;
116 }
117 dsdxj_numerator_part1 *= denominator_s;
118 dsdxj_numerator_part2 *= numerator_s;
119 if ((dsdxj_numerator_part1 - dsdxj_numerator_part2).norm() < std::numeric_limits<scalar_type>::min()) {
120 dsdx[j_elem] = 0;
121 } else {
122 dsdx[j_elem] = (dsdxj_numerator_part1 - dsdxj_numerator_part2) / (denominator_s * denominator_s) * normalization_factor;
123 }
124 dzdx[j_elem] = -1.0 / lambda * dzdxj_numerator / denominator_s;
125 }
126}

130 scalar_type mean_square_displacements = 0.0;

132 cvm::log(std::string("Distance between frame ") + cvm::to_str(i_frame) + " and " + cvm::to_str(i_frame + 1) + " is " + cvm::to_str(rmsd_between_refs[i_frame - 1]) + std::string("\n"));
133 mean_square_displacements += rmsd_between_refs[i_frame - 1] * rmsd_between_refs[i_frame - 1];
134 }
135 mean_square_displacements /= scalar_type(total_frames - 1);
