69 template <typename element_type, typename scalar_type, path_sz path_type>
70 void ArithmeticPathBase<element_type, scalar_type, path_type>::computeValue() {
71  updateDistanceToReferenceFrames();
72  numerator_s = scalar_type(0);
73  denominator_s = scalar_type(0);
74  for (size_t i_frame = 0; i_frame < frame_element_distances.size(); ++i_frame) {
75  scalar_type exponent_tmp = scalar_type(0);
76  for (size_t j_elem = 0; j_elem < num_elements; ++j_elem) {
77  exponent_tmp += weights[j_elem] * frame_element_distances[i_frame][j_elem] * weights[j_elem] * frame_element_distances[i_frame][j_elem];
78  }
79  exponent_tmp = exponent_tmp * -1.0 * lambda;
80  // prevent underflow if the argument of cvm::exp is less than -708.4
81  if (exponent_tmp > -708.4) {
82  exponent_tmp = cvm::exp(exponent_tmp);
83  } else {
84  exponent_tmp = 0;
85  }
86  numerator_s += static_cast<scalar_type>(i_frame) * exponent_tmp;
87  denominator_s += exponent_tmp;
88  s_numerator_frame[i_frame] = static_cast<scalar_type>(i_frame) * exponent_tmp;
89  s_denominator_frame[i_frame] = exponent_tmp;
90  }
91  s = numerator_s / denominator_s * normalization_factor;
92  z = -1.0 / lambda * cvm::logn(denominator_s);
93 }

104  element_type dsdxj_numerator_part1(dsdx[j_elem]);
105  element_type dsdxj_numerator_part2(dsdx[j_elem]);
106  element_type dzdxj_numerator(dsdx[j_elem]);
107  dsdxj_numerator_part1.reset();
108  dsdxj_numerator_part2.reset();
109  dzdxj_numerator.reset();
110  for (size_t i_frame = 0; i_frame < frame_element_distances.size(); ++i_frame) {
111  element_type derivative_tmp = -2.0 * lambda * weights[j_elem] * weights[j_elem] * frame_element_distances[i_frame][j_elem];
112  dsdxj_numerator_part1 += s_numerator_frame[i_frame] * derivative_tmp;
113  dsdxj_numerator_part2 += s_denominator_frame[i_frame] * derivative_tmp;
114  dzdxj_numerator += s_denominator_frame[i_frame] * derivative_tmp;
115  }
116  dsdxj_numerator_part1 *= denominator_s;
117  dsdxj_numerator_part2 *= numerator_s;
118  if ((dsdxj_numerator_part1 - dsdxj_numerator_part2).norm() < std::numeric_limits<scalar_type>::min()) {
119  dsdx[j_elem] = 0;
120  } else {
121  dsdx[j_elem] = (dsdxj_numerator_part1 - dsdxj_numerator_part2) / (denominator_s * denominator_s) * normalization_factor;
122  }
123  dzdx[j_elem] = -1.0 / lambda * dzdxj_numerator / denominator_s;
124  }
