426 inline int colvarvalue::check_types(colvarvalue const &x1,
427  colvarvalue const &x2)
428 {
429  if (!colvarvalue::type_checking()) {
430  return COLVARS_OK;
431  }
432 
433  if (x1.type() != x2.type()) {
434  if (((x1.type() == type_unit3vector) &&
435  (x2.type() == type_unit3vectorderiv)) ||
436  ((x2.type() == type_unit3vector) &&
437  (x1.type() == type_unit3vectorderiv)) ||
438  ((x1.type() == type_quaternion) &&
439  (x2.type() == type_quaternionderiv)) ||
440  ((x2.type() == type_quaternion) &&
441  (x1.type() == type_quaternionderiv))) {
442  return COLVARS_OK;
443  } else {
444  cvm::error("Trying to perform an operation between two colvar "
445  "values with different types, \""+
446  colvarvalue::type_desc(x1.type())+
447  "\" and \""+
448  colvarvalue::type_desc(x2.type())+
449  "\".\n");
450  return COLVARS_ERROR;
451  }
452  }
453 
454  if (x1.type() == type_vector) {
455  if (x1.vector1d_value.size() != x2.vector1d_value.size()) {
456  cvm::error("Trying to perform an operation between two vector colvar "
457  "values with different sizes, "+
458  cvm::to_str(x1.vector1d_value.size())+
459  " and "+
460  cvm::to_str(x2.vector1d_value.size())+
461  ".\n");
462  return COLVARS_ERROR;
463  }
464  }
465  return COLVARS_OK;
466 }

657 inline cvm::real colvarvalue::norm2() const
658 {
659  switch (value_type) {
660  case colvarvalue::type_scalar:
661  return (this->real_value)*(this->real_value);
662  case colvarvalue::type_3vector:
663  case colvarvalue::type_unit3vector:
664  case colvarvalue::type_unit3vectorderiv:
665  return (this->rvector_value).norm2();
666  case colvarvalue::type_quaternion:
667  case colvarvalue::type_quaternionderiv:
668  return (this->quaternion_value).norm2();
669  case colvarvalue::type_vector:
670  if (elem_types.size() > 0) {
671  // if we have information about non-scalar types, use it
672  cvm::real result = 0.0;
673  size_t i;
674  for (i = 0; i < elem_types.size(); i++) {
675  result += (this->get_elem(i)).norm2();
676  }
677  return result;
678  } else {
679  return vector1d_value.norm2();
680  }
681  break;
682  case colvarvalue::type_notset:
683  default:
684  return 0.0;
685  }
686 }

712 inline cvm::real colvarvalue::dist2(colvarvalue const &x2) const
713 {
714  colvarvalue::check_types(*this, x2);
715 
716  switch (this->type()) {
717  case colvarvalue::type_scalar:
718  return (this->real_value - x2.real_value)*(this->real_value - x2.real_value);
719  case colvarvalue::type_3vector:
720  return (this->rvector_value - x2.rvector_value).norm2();
721  case colvarvalue::type_unit3vector:
722  case colvarvalue::type_unit3vectorderiv:
723  // angle between (*this) and x2 is the distance
724  return std::acos(this->rvector_value * x2.rvector_value) * std::acos(this->rvector_value * x2.rvector_value);
725  case colvarvalue::type_quaternion:
726  case colvarvalue::type_quaternionderiv:
727  // angle between (*this) and x2 is the distance, the quaternion
728  // object has it implemented internally
729  return this->quaternion_value.dist2(x2.quaternion_value);
730  case colvarvalue::type_vector:
731  return (this->vector1d_value - x2.vector1d_value).norm2();
732  case colvarvalue::type_notset:
733  default:
734  this->undef_op();
735  return 0.0;
736  };
737 }
