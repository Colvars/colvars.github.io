373 template<colvarscript::Object_type T>
374 int colvarscript::check_cmd_nargs(char const *cmd,
375  int objc,
376  int n_args_min,
377  int n_args_max)
378 {
379  int const shift = cmd_arg_shift<T>();
380  if (objc < shift+n_args_min) {
381  add_error_msg("Insufficient number of arguments ("+cvm::to_str(objc)+
382  ") for script function \""+std::string(cmd)+
383  "\":\n"+get_command_full_help(cmd));
384  return COLVARSCRIPT_ERROR;
385  }
386  if (objc > shift+n_args_max) {
387  add_error_msg("Too many arguments ("+cvm::to_str(objc)+
388  ") for script function \""+std::string(cmd)+
389  "\":\n"+get_command_full_help(cmd));
390  return COLVARSCRIPT_ERROR;
391  }
392  return COLVARSCRIPT_OK;
393 }

423 template<colvarscript::Object_type T>
424 int colvarscript::cmd_arg_shift()
425 {
426  int shift = 0;
427  if (T == use_module) {
428  // "cv" and "COMMAND" are 1st and 2nd argument, and shift is equal to 2
429  shift = 2;
430  } else if (T == use_colvar) {
431  // Same as above with additional arguments "colvar" and "NAME"
432  shift = 4;
433  } else if (T == use_bias) {
434  shift = 4;
435  }
436  return shift;
437 }
