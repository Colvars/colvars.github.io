300 template<colvarscript::Object_type T>
301 int colvarscript::check_cmd_nargs(char const *cmd,
302  int objc,
303  int n_args_min,
304  int n_args_max)
305 {
306  int const shift = cmd_arg_shift<T>();
307  if (objc < shift+n_args_min) {
308  add_error_msg("Missing arguments for script function \""+std::string(cmd)+
309  "\":\n"+get_command_help(cmd));
310  return COLVARSCRIPT_ERROR;
311  }
312  if (objc > shift+n_args_max) {
313  add_error_msg("Too many arguments for script function \""+std::string(cmd)+
314  "\":\n"+get_command_help(cmd));
315  return COLVARSCRIPT_ERROR;
316  }
317  return COLVARSCRIPT_OK;
318 }

348 template<colvarscript::Object_type T>
349 int colvarscript::cmd_arg_shift()
350 {
351  int shift = 0;
352  if (T == use_module) {
353  // "cv" and "COMMAND" are 1st and 2nd argument, and shift is equal to 2
354  shift = 2;
355  } else if (T == use_colvar) {
356  // Same as above with additional arguments "colvar" and "NAME"
357  shift = 4;
358  } else if (T == use_bias) {
359  shift = 4;
360  }
361  return shift;
362 }
