192  script->clear_results(); \
193  if (objc < 2+N_ARGS_MIN) /* "cv" and "COMM" are 1st and 2nd */ { \

199  script->set_str_result("Too many arguments\n" + \

201  return COLVARSCRIPT_ERROR; \
202  } \
203  FN_BODY; \
204  }
