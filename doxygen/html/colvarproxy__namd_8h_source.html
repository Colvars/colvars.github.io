300 inline cvm::real colvarproxy_namd::position_dist2(cvm::atom_pos const &pos1,
301  cvm::atom_pos const &pos2)
302 {
303  Lattice const *l = this->lattice;
304  Vector const p1(pos1.x, pos1.y, pos1.z);
305  Vector const p2(pos2.x, pos2.y, pos2.z);
306  Vector const d = l->delta(p1, p2);
307  return cvm::real(d.x*d.x + d.y*d.y + d.z*d.z);
